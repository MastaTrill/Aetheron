        createdAt: Date.now(),
        userAddress,
        highestPrice: ethers.parseUnits('0', 6)
      };
 
      this.orders.set(orderId, order);
      this.emit('trailingStopCreated', order);
      return order;
    }
 
    throw new Error('Unsupported order type');
  }
}
